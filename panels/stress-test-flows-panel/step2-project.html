Create Dataset with faasProxy Driver

# Create the dataset for storing link items dataset = project.datasets.create( dataset_name='benchmark-images', driver='faasProxy' # Use faasProxy storage driver ) print(f"Dataset created: {dataset.name} ({dataset.id})")

faasProxy Driver: This driver allows the dataset to reference files stored in the FaaS Filestore without copying them, using link items.

Complete Script

import dtlpy as dl # Configuration ORG_ID = 'YOUR_ORG_ID' PROJECT_NAME = 'nginx-stress-test' DATASET_NAME = 'benchmark-images' # Login dl.login() # Get organization org = dl.organizations.get(organization_id=ORG_ID) print(f"Organization: {org.name}") # Create project try: project = dl.projects.get(project_name=PROJECT_NAME) print(f"Project exists: {project.name}") except dl.exceptions.NotFound: project = dl.projects.create( project_name=PROJECT_NAME, org_id=org.id ) print(f"Project created: {project.name}") # Create dataset with faasProxy driver try: dataset = project.datasets.get(dataset_name=DATASET_NAME) print(f"Dataset exists: {dataset.name}") except dl.exceptions.NotFound: dataset = project.datasets.create( dataset_name=DATASET_NAME, driver='faasProxy' ) print(f"Dataset created: {dataset.name}") print(f"\nProject ID: {project.id}") print(f"Dataset ID: {dataset.id}")
