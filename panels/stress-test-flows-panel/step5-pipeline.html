import dtlpy as dl # Configuration PROJECT_NAME = 'nginx-stress-test' DATASET_NAME = 'benchmark-images' PIPELINE_NAME = 'stress-test-sam2-pipeline' # Login dl.login() project = dl.projects.get(project_name=PROJECT_NAME) dataset = project.datasets.get(dataset_name=DATASET_NAME) # Create pipeline pipeline = project.pipelines.create(name=PIPELINE_NAME) # Add dataset node (trigger) dataset_node = pipeline.nodes.add( node_type=dl.DatasetNode, name='Input Images', dataset_id=dataset.id ) # Add SAM2 function node (TBD - get details from Or) # sam2_node = pipeline.nodes.add( # node_type=dl.FunctionNode, # name='SAM2 Segmentation', # package_name='sam2-dpk', # module_name='main', # function_name='segment' # ) # Connect nodes # pipeline.connections.add( # source=dataset_node, # target=sam2_node # ) # Install and start pipeline pipeline.install() pipeline = pipeline.open_in_web() print(f"Pipeline created: {pipeline.name} ({pipeline.id})")
